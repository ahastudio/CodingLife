```bash
npm init -y

npm install --save-dev eslint
npx eslint --init

npm install --save-dev typescript
npx tsc --init

npm install react react-dom
npm install --save-dev @types/react @types/react-dom

npm install --save-dev esbuild livereload concurrently

npm run start
```
